# Interactive Map (v1 core) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **NOTE — two spikes are run in the MAIN session, not by subagents:** Task 1 (chunk-index decode) needs live `location_state` sampling and Task 2 (base-image extraction) needs the local BitCraft install. The orchestrator runs these and records the results into the plan before dispatching the dependent tasks.

**Goal:** An interactive `/map` (Leaflet + CRS.Simple) over an extracted base terrain image, showing claims, the region grid, and empire territory from our own auto-refreshing data.

**Architecture:** Game `x,z` (0–23040) map directly to Leaflet `CRS.Simple` (the bitcraftmap pattern). The worker snapshot gains map tables; pure mappers (incl. a sampled `chunkIndex→x,z` decode) feed compact per-layer payloads; a client-island Leaflet map renders them over a base image extracted from the install. World bounds/region grid derive live from `world_region_state`, so new regions appear automatically.

**Tech Stack:** Next.js 16, react-leaflet + leaflet, Drizzle/Neon, `ws` SpacetimeDB, UnityPy (Python), Vitest. Spec: `docs/superpowers/specs/2026-06-06-phase-4-map-design.md`.

```ts
// packages/shared/src/world/coords.ts
// CHUNK_SIZE + the index layout are derived from sampled location_state (Task 1 Step 1).
// Pin the resolved constant here. If the geometry is hex/non-uniform and an exact decode
// is intractable, ship regions + claims and FAST-FOLLOW territory (see spec §8).
export const CHUNK_SIZE = /* derived, e.g. 60 */ 60;
export const WORLD_CHUNK_STRIDE = /* derived chunks-per-row, e.g. region_count_sqrt*width */ 0;

export interface Bounds { x0: number; z0: number; x1: number; z1: number; }

export function chunkIndexToBounds(chunkIndex: string | number): Bounds {
  const idx = typeof chunkIndex === "string" ? Number(chunkIndex) : chunkIndex;
  const cx = idx % WORLD_CHUNK_STRIDE;
  const cz = Math.floor(idx / WORLD_CHUNK_STRIDE);
  const x0 = cx * CHUNK_SIZE, z0 = cz * CHUNK_SIZE;
  return { x0, z0, x1: x0 + CHUNK_SIZE, z1: z0 + CHUNK_SIZE };
}

export function regionBounds(r: { minChunkX: number; minChunkZ: number; widthChunks: number; heightChunks: number }): Bounds {
  return {
    x0: r.minChunkX * CHUNK_SIZE,
    z0: r.minChunkZ * CHUNK_SIZE,
    x1: (r.minChunkX + r.widthChunks) * CHUNK_SIZE,
    z1: (r.minChunkZ + r.heightChunks) * CHUNK_SIZE,
  };
}

type Raw = Record<string, unknown>;
export function decodeLocationSum(loc: unknown): { x: number; z: number; dimension: number } | null {
  // Live form: positional Sum [tag, {x,z,dimension}]; also accept the keyed object.
  const payload = Array.isArray(loc) ? loc[1] : loc;
  if (payload && typeof payload === "object") {
    const o = payload as Raw;
    if (typeof o.x === "number" && typeof o.z === "number") {
      return { x: o.x, z: o.z, dimension: typeof o.dimension === "number" ? o.dimension : 1 };
    }
  }
  return null;
}
```

// packages/shared/src/ingest/map-world.ts
import { toInt } from "./decode";
import { decodeLocationSum } from "../world/coords";

type Raw = Record<string, unknown>;
const str = (v: unknown) => (typeof v === "string" ? v : v == null ? "" : String(v));
const idStr = (v: unknown) => (v == null ? "" : String(v));

export interface MapClaimRow { entityId: string; name: string; x: number; z: number; dimension: number; numTiles: number; treasury: number; supplies: number; }
export function mapClaimLocalRows(rows: Raw[], nameById: Map<string, string>): MapClaimRow[] {
  const out: MapClaimRow[] = [];
  for (const r of rows) {
    const loc = decodeLocationSum(r.location);
    if (!loc) continue;
    const id = idStr(r.entity_id);
    out.push({
      entityId: id,
      name: nameById.get(id) ?? `Claim ${id}`,
      x: loc.x, z: loc.z, dimension: loc.dimension,
      numTiles: toInt(r.num_tiles) ?? 0,
      treasury: toInt(r.treasury) ?? 0,
      supplies: toInt(r.supplies) ?? 0,
    });
  }
  return out;
}

export interface MapChunkRow { chunkIndex: string; empireEntityId: string; watchtowerEntityId: string | null; }
export function mapChunkRows(rows: Raw[]): MapChunkRow[] {
  return rows.map((r) => ({
    chunkIndex: idStr(r.chunk_index),
    empireEntityId: idStr(r.empire_entity_id),
    watchtowerEntityId: r.watchtower_entity_id != null ? idStr(r.watchtower_entity_id) : null,
  }));
}

export interface MapRegionRow { id: number; name: string | null; minChunkX: number; minChunkZ: number; widthChunks: number; heightChunks: number; regionIndex: number; }
export function mapRegionRows(rows: Raw[], nameById: Map<number, string>): MapRegionRow[] {
  return rows.map((r) => {
    const id = toInt(r.id)!;
    return {
      id,
      name: nameById.get(id) ?? null,
      minChunkX: toInt(r.region_min_chunk_x) ?? 0,
      minChunkZ: toInt(r.region_min_chunk_z) ?? 0,
      widthChunks: toInt(r.region_width_chunks) ?? 0,
      heightChunks: toInt(r.region_height_chunks) ?? 0,
      regionIndex: toInt(r.region_index) ?? 0,
    };
  });
}
```

// apps/web/lib/queries/map.ts
import "server-only";
import { getDb, schema } from "@/lib/db";
import { chunkIndexToBounds } from "@bcc/shared";

export interface ClaimPoint { id: string; name: string; x: number; z: number; tiles: number; treasury: number; }
export interface RegionRect { id: number; name: string | null; x0: number; z0: number; x1: number; z1: number; }
export interface TerritoryCell { x0: number; z0: number; empire: string; }

export async function getMapClaims(): Promise<ClaimPoint[]> {
  const rows = await getDb().select().from(schema.mapClaims);
  return rows.map((c) => ({ id: c.entityId, name: c.name, x: c.x, z: c.z, tiles: c.numTiles, treasury: Number(c.treasury) }));
}

export async function getMapRegions(): Promise<RegionRect[]> {
  const rows = await getDb().select().from(schema.mapRegions);
  // CHUNK_SIZE applied via regionBounds in shared; inline here using chunk→x,z from coords.
  const { regionBounds } = await import("@bcc/shared");
  return rows.map((g) => {
    const b = regionBounds({ minChunkX: g.minChunkX, minChunkZ: g.minChunkZ, widthChunks: g.widthChunks, heightChunks: g.heightChunks });
    return { id: g.id, name: g.name, x0: b.x0, z0: b.z0, x1: b.x1, z1: b.z1 };
  });
}

export async function getTerritoryCells(): Promise<TerritoryCell[]> {
  const rows = await getDb().select().from(schema.mapChunks);
  return rows.map((c) => {
    const b = chunkIndexToBounds(c.chunkIndex);
    return { x0: b.x0, z0: b.z0, empire: c.empireEntityId };
  });
}
```
